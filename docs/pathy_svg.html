 208        """Lazy-build index of id -> element for O(1) lookup.
 209
 210        Note: this cache is NOT invalidated when the tree is mutated.

 318    def xy_guide(self, *, color: str = "red", step: float = 50) -> SVGDocument:
 319        """Return a copy with a coordinate grid overlay for orientation."""
 320        clone = self._clone()
 321        vb = clone.viewbox
 322        if vb is None:
 323            return clone
 324
 325        root = clone.root
 326        ns = root.nsmap.get(None, SVG_NS)
 327        g = etree.SubElement(root, f"{{{ns}}}g" if ns else "g", id="pathy-guide")
 328        g.set("style", f"stroke:{color};stroke-width:0.5;fill:none;opacity:0.5")
 329
 330        x = vb.x
 331        while x <= vb.x + vb.width:
 332            line = etree.SubElement(g, f"{{{ns}}}line" if ns else "line")
 333            line.set("x1", str(x))
 334            line.set("y1", str(vb.y))
 335            line.set("x2", str(x))
 336            line.set("y2", str(vb.y + vb.height))
 337            # Label
 338            txt = etree.SubElement(g, f"{{{ns}}}text" if ns else "text")
 339            txt.set("x", str(x + 2))
 340            txt.set("y", str(vb.y + 12))
 341            txt.set("style", f"fill:{color};font-size:8px;stroke:none")
 342            txt.text = str(int(x))
 343            x += step
 344
 345        y = vb.y
 346        while y <= vb.y + vb.height:
 347            line = etree.SubElement(g, f"{{{ns}}}line" if ns else "line")
 348            line.set("x1", str(vb.x))
 349            line.set("y1", str(y))
 350            line.set("x2", str(vb.x + vb.width))
 351            line.set("y2", str(y))
 352            txt = etree.SubElement(g, f"{{{ns}}}text" if ns else "text")
 353            txt.set("x", str(vb.x + 2))
 354            txt.set("y", str(y - 2))
 355            txt.set("style", f"fill:{color};font-size:8px;stroke:none")
 356            txt.text = str(int(y))
 357            y += step
 358
 359        return clone

 584
 585        clone = self._clone()
 586        vb = clone.viewbox
 587        if vb is None:
 588            from pathy_svg.transform import ViewBox
 589

 591
 592        scale = getattr(self, "_last_scale", None)
 593        cat_pal = getattr(self, "_last_categorical_palette", None)
 594
 595        if kind == "auto":
 596            if cat_pal is not None:
 597                kind = "categorical"
 598            elif scale is not None and scale.breaks is not None:
 599                kind = "discrete"
 600            else:
 601                kind = "gradient"
 602
 603        if kind == "gradient" and scale is not None:
 604            legend_elem = build_gradient_legend(
